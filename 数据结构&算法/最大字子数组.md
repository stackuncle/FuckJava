public class Solution {
    public int maxSubArray(int[] A) {
        if (A.length == 0) return 0;
        
        int currMax = A[0], resMax = A[0];
        for (int i = 1; i < A.length; i++) {
            currMax = Math.max(A[i], currMax + A[i]);
            resMax = Math.max(currMax, resMax);
        }
        
        return resMax;
    }
}
